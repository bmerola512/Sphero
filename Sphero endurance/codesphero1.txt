async function startProgram() {
	setMainLed({ r: 81, g: 255, b: 107 });
	await speak('ready, set, go', true);
	setStabilization(true);
	await roll(0, 82, 9.35);
	stopRoll();
	await delay(1.3);
	await roll(90, 70, 6.5);
	stopRoll();
	await delay(1.3);
	await roll(180, 83, 9.34);
	stopRoll();
	await delay(0);
	await roll(270, 70, 6.5);
	stopRoll();
	await speak('I\'m done, and I need water', true);
	setMainLed({ r: 255, g: 34, b: 51 });
}
